\lim_{n \to \infty}\sum_{i=1}^n{\frac{1}{n}}\sin\frac{k}{n}

   g(x,y) = \left\{\begin{array}{ll}
      f(x,y), & \mbox{if $x<y$} \\  % 文字的部份要用 \mbox

      0,      & \mbox{otherwise.}
     \end{array} \right.
\end{equation}
